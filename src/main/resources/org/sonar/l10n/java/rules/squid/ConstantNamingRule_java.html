这个规则检测所有常量，常量命名必须全大写，单词之间用下划线隔开

Noncompliant Code Example

public class ConstantNamingRule {

    public static final String name = "Tom"; // Noncompliant

    public static final Integer max = 1; // Noncompliant

    public static final boolean success = true; // Noncompliant

    public static final long serialVersionUID = 1L;

    public static final Logger logger = new Logger();

}

Compliant Solution

public class ConstantNamingRule {

    public static final String NAME = "Tom";

    public static final Integer MAX = 1;

    public static final boolean SUCCESS = true;

    public static final long serialVersionUID = 1L;

    public static final Logger logger = new Logger();

}
